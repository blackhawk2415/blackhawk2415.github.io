                            from xlwings import Workbook, Range
                            import requests, logging, datetime, time, random, os


                            # Initiate Logging
                            logging.basicConfig(filename='errors.log', level=logging.INFO)

                            # Global Variables
                            jsonData = {}
                            outputData = []
                            now = datetime.datetime.now().strftime("%m-%d-%Y %H:%M")


                            # Establish Connection Between Workbook and Python

                            def connect():
                                wb = Workbook.caller()
                                Range("Dashboard", "excelData").clear_contents()
                                logging.critical(now + ": Could not connect with Workbook, check VBA connection (Alt+F11)")
                                query()


                            # Making ReST Query to Yahoo and Retrieving Data
                            def query():
                                queryString = ""
                                counter = 1
                                stockList = Range('A6:A20').value
                                # stockList = ["MSFT", "GOOG", "YAHOO"] # Debugging Purposes
                                for stock in stockList:
                                    queryString += '"' + str(stock) + '"' + '%2C'

                                queryString = queryString[0:-3]
                                stringAddon = "store%3A%2F%2Fdatatables.org%2Falltableswithkeys&callback="
                                query = "https://query.yahooapis.com/v1/public/yql?q=select * from yahoo.finance.quotes where symbol in (%s)&format=json&diagnostics=true&env=%s" % (queryString, stringAddon,)
                                logging.info("ReST Query Made at " + now)
                                # print query
                                r = requests.get(query)

                                if r.status_code == 200:
                                    jsonData = r.json()
                                    logging.debug(now + "Successful Request Made to Yahoo")
                                    # print jsonData
                                else:
                                    while counter < 5:
                                        logging.warning(now + ": ReST query failed, check site URL and server status. Attempt #" + counter)
                                        r = requests.get(query)
                                        if r.status_code == 200:
                                            jsonData = r.json()
                                            break
                                        counter += 1
                                        time.sleep(random.randrange(2, int(counter)*5))


                                counter = 0 #reset counter

                                logging.info(now + "Began parsing JSON response")
                                while counter < len(stockList): 
                                    # Parsing JSON response into actionable data
                                    stockDataRaw = [
                                        str(jsonData["query"]["results"]["quote"][counter]["AverageDailyVolume"]),
                                        str(jsonData["query"]["results"]["quote"][counter]["Ask"]),
                                        str(jsonData["query"]["results"]["quote"][counter]["Change"]),
                                        str(jsonData["query"]["results"]["quote"][counter]["BookValue"]),
                                        str(jsonData["query"]["results"]["quote"][counter]["EarningsShare"]),
                                        str(jsonData["query"]["results"]["quote"][counter]["MarketCapitalization"]),
                                        str(jsonData["query"]["results"]["quote"][counter]["EBITDA"]),
                                        str(jsonData["query"]["results"]["quote"][counter]["PERatio"]),
                                        str(jsonData["query"]["results"]["quote"][counter]["PEGRatio"]),
                                    ]
                                    StockData = stockDataRaw
                                    outputData.append(StockData)
                                    counter += 1

                                # Writing data to Excel Workbook    
                                Range("Dashboard", "excelData").value = outputData


                            if __name__ == '__main__':
                                path = os.path.abspath(os.path.join(os.path.dirname(__file__), 'SmallCapValueStocks.xlsm'))
                                Workbook.set_mock_caller(path)

                                connect()
